<script>
  function getRandomColor() {
      var letters = '0123456789ABCDEF'.split('');
      var color = '#';
      for (var i = 0; i < 6; i++ ) {
          color += letters[Math.floor(Math.random() * 16)];
      }
      return color;
  }

  const chart = new Chart(document.getElementById("chart"), {
    type: 'line',
    data: {
      label: 'test',
      datasets: [
        {#for item in items} {
          label: '{item.uniqueId}',
          link: '{item.link}',
          data: [
            {#for data in item.data} {
              t: "{data.timeFormatted}",
              y: {data.price}
            }, {/for}
          ],
          fill: false,
          borderColor: getRandomColor()
        }, {/for}
      ]
    },
    options: {
      scales: {
        xAxes: [{
          type: 'time',
          distribution: 'linear',
          time: {
            unit: 'hour',
            timezone: 'Europe/Tallinn',
            displayFormats: {
              hour: 'DD.MM HH.mm'
            }
          }
        }],
        yAxes: [{
          ticks: {
            min: {minPrice},
            max: {maxPrice},
            stepSize: 100
          }
        }]
      },
      legend: {
        display: false
      },
      title: {
        display: true,
        text: '{title}'
      },
      tooltips: {
        callbacks: {
           label: function(item, data) {
             const name = data.datasets[item.datasetIndex].label;
             const price = item.yLabel;
             return name + ": " + price + " EUR";
           }
        }
      },
      onClick: function(event, data) {
        const elements = chart.getElementAtEvent(event);
        if (elements.length > 0) {
          const datasetIndex = elements[0]._datasetIndex;
          const link = chart.data.datasets[datasetIndex].link;
          window.open(link);
        }
      }
    }
  })
</script>
